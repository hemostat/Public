out <- lapply(N, function(n) {
  par <- gxefam_par(num_fam = n / 5, sim_gen = "rbinom", sim_phen = "no_rand",
    freq_g  = p)
  
  prod <- sapply(seq(1, S), function(s) {
    xg <- gxefam_sim_gen(par)
    xg_c <- xg - mean(xg)
    
    as.numeric(crossprod(xg_c))

})

tab <- bind_rows(out)

2.4 Genetic variable in related: not centered

\(E(x_c^T x_c) = tr(V) + \mu^T \mu\)

The first term:

  • \(V = \sigma^2_x K\)
  • \(\sigma^2_x = 2 p q\)
  • \(tr(K) = N\) (all diagonal entries of \(K\) are 1)

The second term:

  • \(\mu = 2 p\)
  • \(\mu^T \mu = (2p)^2\)
out <- lapply(N, function(n) {

    freq_g  = p)
  
  prod <- sapply(seq(1, S), function(s) {
    xg <- gxefam_sim_gen(par)
    
    as.numeric(crossprod(xg))

})

  mutate(

3 Stratified genetic variable

3.1 Genetic variable in unrelated: centered; stratified

\(x_c \sim (0, V) = (0, \sigma_x^2 I)\)

\(y = A x_c \sim (0, A^T V A)\)

Then:

\(y = diag(a) x_c = A x_c \sim (0, A^T V A) = (0, \sigma_x^2 A^T A) = (0, \sigma_x^2 diag(a^2))\)

Since elements in \(a\) are 0 and 1:

\(y \sim (0, \sigma_x^2 diag(a)) = (0, V_y)\)

Further:

\(E(y^T y) = tr(V_y) = f \sigma_x^2 N = 2 p q f N\)

out <- lapply(N, function(n) {
  par <- gxefam_par(num_fam = n / 5, sim_gen = "rbinom", sim_phen = "no_rand",
    freq_g  = p)
  
  prod <- sapply(seq(1, S), function(s) {
    xg <- gxefam_sim_gen(par)
    xg_c <- xg - mean(xg)
    
    xs <- c(rep(1, f * n), rep(0, (1 - f) * n))
    xgs <- as.numeric(diag(xs) %*% xg_c)
    
    as.numeric(crossprod(xgs))
  })
  
  data_frame(N = as.integer(n), prod = prod)
})

tab <- bind_rows(out)

tab_expected <- data_frame(N = N) %>%
  mutate(
    prod = 2 * p * (1 - p) * f * N)

3.2 Genetic variable in unrelated: centered; stratified by centered

\(x_c \sim (0, V) = (0, \sigma_x^2 I)\)

\(y = A x_c \sim (0, A^T V A)\)

Then:

\(y = a_c x_c = diag(a - \mu_a) x_c = A_c x_c \sim (0, A_c^T V A_c) = (0, \sigma_x^2 A_c^T A_c) = (0, \sigma_x^2 diag((a - \mu_a)^2)) = (0, \sigma_x^2 diag(a_c^2))\)

Further:

\(a_c^2 = (a - \mu_a)^2 = a^2 - 2 \mu_a a + \mu_a^2 = a - 2 \mu_a a + \mu_a^2 = (1 - 2 \mu_a) a + \mu_a^2\)

\(\mu_{a_c^2} = (1 - \mu_a) \mu_a + \mu_a^2 = \mu_a - \mu_a^2 = \mu_a (1 - \mu_a)\)

Next:

\(E(y^T y) = tr(V_y) = \sigma_x^2 \mu_{a_c^2} N = 2 p q f (1 - f) N\)
